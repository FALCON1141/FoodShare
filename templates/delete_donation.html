@app.route('/delete_donation/') def delete_donation(id): if 'user_id' not in session: return redirect(url_for('login')) donation = Donation.query.get(id) if donation is None: return "Donation not found" if donation.donor_id != session['user_id']: return "Unauthorized" db.session.delete(donation) db.session.commit() return redirect(url_for('donation_history'))
